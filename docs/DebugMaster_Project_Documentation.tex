\section{Introduction}

\subsection{Project Overview}

DebugMaster is an Android mobile game designed to help developers improve their debugging skills through gamified challenges. Players solve buggy code snippets across multiple programming languages, earning XP and unlocking achievements.

\begin{infobox}[Key Features]
\begin{itemize}[leftmargin=*]
    \item \textbf{7 Programming Languages:} Java, Python, JavaScript, C++, C\#, Swift, Kotlin
    \item \textbf{Multiple Game Modes:} Practice, Daily Challenge, Battle Arena, Learning Paths
    \item \textbf{Gamification:} XP system, streaks, achievements, leaderboards
    \item \textbf{Social Features:} Friends system, shareable results
    \item \textbf{Customization:} Editor themes, fonts, mascot
\end{itemize}
\end{infobox}

\subsection{Technology Stack}

\begin{center}
\begin{tabular}{lll}
\toprule
\textbf{Component} & \textbf{Technology} & \textbf{Purpose} \\
\midrule
Language & Java & Primary development language \\
UI Framework & Android Views + Data Binding & User interface \\
Architecture & MVVM & Clean separation of concerns \\
Database & Room (SQLite) & Local data persistence \\
DI & Hilt (Dagger) & Dependency injection \\
Navigation & Jetpack Navigation & Screen navigation \\
Build & Gradle (Kotlin DSL) & Build system \\
\bottomrule
\end{tabular}
\end{center}

\section{Getting Started}

\subsection{Prerequisites}

Before you begin, ensure you have the following installed:

\begin{itemize}[leftmargin=*]
    \item \textbf{Android Studio:} Arctic Fox (2020.3.1) or newer
    \item \textbf{JDK:} Java 11 or higher
    \item \textbf{Android SDK:} API Level 24 (Android 7.0) minimum
    \item \textbf{Git:} Version 2.0 or higher
    \item \textbf{Gradle:} 8.0+ (included with project)
\end{itemize}

\subsection{Cloning the Repository}

\begin{codebox}[Clone Commands]
\begin{lstlisting}[style=bash]
# Clone the repository
git clone https://github.com/hamdanyasser/DebugAppProject.git

# Navigate to project directory
cd DebugAppProject

# Checkout the feature branch with all updates
git checkout claude/fix-game-bugs-01A7KyWVzeYiWXRn7MbR56sA

# Verify you're on the correct branch
git branch
\end{lstlisting}
\end{codebox}

\subsection{Opening in Android Studio}

\begin{enumerate}[leftmargin=*]
    \item Open Android Studio
    \item Select \textbf{File ‚Üí Open}
    \item Navigate to the cloned \texttt{DebugAppProject} folder
    \item Click \textbf{OK} to open the project
    \item Wait for Gradle sync to complete (may take 2-5 minutes first time)
    \item If prompted, install any missing SDK components
\end{enumerate}

\begin{warningbox}[Common Issues]
\textbf{Gradle Sync Failed:}
\begin{itemize}
    \item Ensure you have JDK 11+ set in Project Structure
    \item Try \textbf{File ‚Üí Invalidate Caches / Restart}
    \item Check your internet connection for dependency downloads
\end{itemize}
\end{warningbox}

\subsection{Running the App}

\begin{enumerate}[leftmargin=*]
    \item Connect an Android device (USB debugging enabled) or start an emulator
    \item Select your device from the device dropdown in the toolbar
    \item Click the \textbf{Run} button (green play icon) or press \texttt{Shift+F10}
    \item Wait for the build and installation to complete
\end{enumerate}

\begin{successbox}[Recommended Emulator Settings]
\begin{itemize}
    \item \textbf{Device:} Pixel 4 or newer
    \item \textbf{API Level:} 30 (Android 11) or higher
    \item \textbf{RAM:} 2GB minimum
    \item \textbf{Storage:} 2GB internal storage
\end{itemize}
\end{successbox}

\subsection{Phase 4: Dopamine Loop System}

\begin{infobox}[CelebrationManager.java - New File]
\textbf{Location:} \texttt{app/src/main/java/.../util/CelebrationManager.java}

\textbf{Features:}
\begin{itemize}
    \item Random celebration messages: "Perfect Fix!", "Bug Annihilated!", "Clean Execution!"
    \item Combo system: 5\% XP bonus per consecutive solve (max 50\%)
    \item 5-minute combo timeout
    \item Pulse and bounce animations
    \item Integration with SoundManager for audio feedback
\end{itemize}

\textbf{Usage:}
\begin{lstlisting}[style=java]
CelebrationManager celebrationManager =
    new CelebrationManager(context);

// Record a solve and get celebration data
CelebrationResult result = celebrationManager.recordSolve(
    isPerfect,      // true if solved without hints
    "Hard",         // difficulty
    50,             // XP earned
    currentStreak   // user's streak
);

// Access celebration data
String message = result.mainMessage;  // "Perfect Fix!"
int combo = result.combo;             // 3
int bonus = result.comboBonus;        // +15 XP
\end{lstlisting}
\end{infobox}

\section{Troubleshooting}

\subsection{Common Issues}

\begin{warningbox}[Build Errors]
\textbf{Problem:} \texttt{Could not resolve dependencies}

\textbf{Solution:}
\begin{enumerate}
    \item Check internet connection
    \item Run \texttt{./gradlew clean}
    \item Run \texttt{./gradlew build --refresh-dependencies}
\end{enumerate}
\end{warningbox}

\begin{warningbox}[Database Issues]
\textbf{Problem:} App crashes on launch with Room error

\textbf{Solution:}
\begin{enumerate}
    \item Uninstall app from device
    \item Clean and rebuild project
    \item Run again to recreate database
\end{enumerate}
\end{warningbox}

\begin{warningbox}[Hilt Errors]
\textbf{Problem:} \texttt{Hilt injection error}

\textbf{Solution:}
\begin{enumerate}
    \item Ensure \texttt{@AndroidEntryPoint} on Activity/Fragment
    \item Check \texttt{@HiltAndroidApp} on Application class
    \item Rebuild project: \texttt{./gradlew clean assembleDebug}
\end{enumerate}
